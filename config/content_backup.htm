def draw_circle():
size = 10
center_x = size // 2
center_y = size // 2
radius = 4
for y in range(size):
for x in range(size):
# 計算到圓心的距離
distance = ((x - center_x) ** 2 + (y - center_y) ** 2) ** 0.5
# 判斷距離是否接近圓形邊緣，若是則繪製圓
if abs(distance - radius) < 1:
print('*', end='')
else:
print(' ', end='')
print()

from browser import html
from browser import document as doc
# 利用html建立畫布
canvas = html.CANVAS(width=400, height=400)
brython_div = doc["brython_div1"]
brython_div <= canvas # 將畫布加入到網頁中
ctx = canvas.getContext("2d") # 獲取畫布的2D繪圖上下文
# 每格的像素數
gs = 20 # 每格的像素大小
grid_size = 10 # 總共 10x10 的格子
# 定義畫矩形的函數
def dRect(lux, luy, w, h, s=1, c='#ff0000'):
ctx.lineWidth = s
ctx.strokeStyle = c
ctx.beginPath()
ctx.rect(lux, luy, w, h)
ctx.stroke()
# 定義繪製網格的函數
def grid(width, height, grid_pix):
for i in range(width):
for j in range(height):
dRect(i * grid_pix, j * grid_pix, grid_pix, grid_pix, 1, "lightgrey")
# 填充顏色的函數
def fill(x, y, color):
ctx.fillStyle = color
ctx.fillRect(x - 1, y - 1, gs, gs)
# 繪製圓形
def draw_circle():
# 設定圓心和半徑
center_x = grid_size // 2 # 圓心的X坐標（5）
center_y = grid_size // 2 # 圓心的Y坐標（5）
radius = 4 # 圓的半徑
# 計算每個格子是否在圓內
for i in range(grid_size):
for j in range(grid_size):
# 計算格子四個角的座標
x_min = i
y_min = j
x_max = i + 1
y_max = j + 1
# 計算格子四個角的距離，檢查是否在圓內
for x in [x_min, x_max]:
for y in [y_min, y_max]:
# 計算格子角落到圓心的距離
distance = ((x - center_x) ** 2 + (y - center_y) ** 2) ** 0.5
if distance <= radius: # 如果距離小於圓的半徑，填充顏色
fill(i * gs, j * gs, "black")
break
# 繪製網格
grid(grid_size, grid_size, gs)
# 繪製圓形
draw_circle()
